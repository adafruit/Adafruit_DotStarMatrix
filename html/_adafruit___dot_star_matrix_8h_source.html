31 // Matrix layout information is passed in the 'matrixType' parameter for
32 // each constructor (the parameter immediately following is the LED type
33 // from Adafruit_DotStar.h).
34 
35 // These define the layout for a single 'unified' matrix (e.g. one made
36 // from DotStar strips), or for the pixels within each matrix of a tiled
37 // display.
38 
39 #define DS_MATRIX_TOP 0x00 // Pixel 0 is at top of matrix
40 #define DS_MATRIX_BOTTOM 0x01 // Pixel 0 is at bottom of matrix
41 #define DS_MATRIX_LEFT 0x00 // Pixel 0 is at left of matrix
42 #define DS_MATRIX_RIGHT 0x02 // Pixel 0 is at right of matrix
43 #define DS_MATRIX_CORNER 0x03 // Bitmask for pixel 0 matrix corner
44 #define DS_MATRIX_ROWS 0x00 // Matrix is row major (horizontal)
45 #define DS_MATRIX_COLUMNS 0x04 // Matrix is column major (vertical)
46 #define DS_MATRIX_AXIS 0x04 // Bitmask for row/column layout
47 #define DS_MATRIX_PROGRESSIVE 0x00 // Same pixel order across each line
48 #define DS_MATRIX_ZIGZAG 0x08 // Pixel order reverses between lines
49 #define DS_MATRIX_SEQUENCE 0x08 // Bitmask for pixel line order
50 
51 // These apply only to tiled displays (multiple matrices):
52 
53 #define DS_TILE_TOP 0x00 // First tile is at top of matrix
54 #define DS_TILE_BOTTOM 0x10 // First tile is at bottom of matrix
55 #define DS_TILE_LEFT 0x00 // First tile is at left of matrix
56 #define DS_TILE_RIGHT 0x20 // First tile is at right of matrix
57 #define DS_TILE_CORNER 0x30 // Bitmask for first tile corner
58 #define DS_TILE_ROWS 0x00 // Tiles ordered in rows
59 #define DS_TILE_COLUMNS 0x40 // Tiles ordered in columns
60 #define DS_TILE_AXIS 0x40 // Bitmask for tile H/V orientation
61 #define DS_TILE_PROGRESSIVE 0x00 // Same tile order across each line
62 #define DS_TILE_ZIGZAG 0x80 // Tile order reverses between lines
63 #define DS_TILE_SEQUENCE 0x80 // Bitmask for tile line order
